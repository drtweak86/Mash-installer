```rust
enum SoftwareTier {

    /// Criteria: Industry standard, excellent documentation, active development
    /// Examples: Brave, VS Code, Git, Docker

    /// Criteria: High quality, good community, may lack some features
    /// Examples: Firefox, Neovim, Podman

}
```
